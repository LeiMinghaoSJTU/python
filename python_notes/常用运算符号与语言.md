# Python 中的常用运算符号和语言特性

## 1. 算术运算符

```python
# 基本算术运算
a = 10
b = 3

print(a + b)   # 加法: 13
print(a - b)   # 减法: 7
print(a * b)   # 乘法: 30
print(a / b)   # 除法: 3.333...
print(a // b)  # 整除: 3
print(a % b)   # 取模: 1
print(a ** b)  # 幂运算: 1000
```

## 2. 比较运算符

```python
x = 5
y = 10

print(x == y)  # 等于: False
print(x != y)  # 不等于: True
print(x > y)   # 大于: False
print(x < y)   # 小于: True
print(x >= y)  # 大于等于: False
print(x <= y)  # 小于等于: True
```

## 3. 逻辑运算符

```python
a = True
b = False

print(a and b)  # 逻辑与: False
print(a or b)   # 逻辑或: True
print(not a)    # 逻辑非: False

# 实际应用
age = 25
has_license = True

can_drive = age >= 18 and has_license
print(can_drive)  # True
```

## 4. 赋值运算符

```python
x = 10
x += 5      # x = x + 5 → 15
x -= 3      # x = x - 3 → 12
x *= 2      # x = x * 2 → 24
x /= 4      # x = x / 4 → 6.0
x //= 2     # x = x // 2 → 3.0
x %= 2      # x = x % 2 → 1.0
x **= 3     # x = x ** 3 → 1.0
```

## 5. 位运算符

```python
a = 10      # 二进制: 1010
b = 4       # 二进制: 0100

print(a & b)   # 按位与: 0 (0000)
print(a | b)   # 按位或: 14 (1110)
print(a ^ b)   # 按位异或: 14 (1110)
print(~a)      # 按位取反: -11
print(a << 1)  # 左移: 20 (10100)
print(a >> 1)  # 右移: 5 (0101)
```

## 6. 成员运算符

```python
fruits = ['apple', 'banana', 'orange']

print('apple' in fruits)     # True
print('grape' not in fruits) # True

# 字符串检查
text = "Hello World"
print('Hello' in text)       # True
```

## 7. 身份运算符

```python
a = [1, 2, 3]
b = [1, 2, 3]
c = a

print(a is b)      # False - 不同对象
print(a is c)      # True - 同一对象
print(a == b)      # True - 值相等
print(id(a))       # 查看内存地址
print(id(b))
```

## 8. 字符串运算符

```python
str1 = "Hello"
str2 = "World"

print(str1 + " " + str2)  # 字符串连接: "Hello World"
print(str1 * 3)           # 字符串重复: "HelloHelloHello"
print('H' in str1)        # 字符检查: True
```

## 9. 三元运算符

```python
# 传统写法
age = 20
if age >= 18:
    status = "成年"
else:
    status = "未成年"

# 三元运算符写法
status = "成年" if age >= 18 else "未成年"
print(status)  # 成年
```

## 10. 海象运算符 (Python 3.8+)

```python
# 传统写法
data = "Hello World"
if len(data) > 5:
    print(f"字符串长度是 {len(data)}")

# 使用海象运算符
if (n := len(data)) > 5:
    print(f"字符串长度是 {n}")

# 在循环中使用
while (line := input("输入内容: ")) != "quit":
    print(f"你输入了: {line}")
```

## 11. 运算符优先级

```python
# 从高到低的优先级示例
result = 2 + 3 * 4        # 14 (乘法优先)
result = (2 + 3) * 4      # 20 (括号优先)
result = 2 ** 3 * 4       # 32 (指数优先)
result = not True or False # False (not优先)
```

## 12. 特殊运算技巧

```python
# 链式比较
x = 5
print(1 < x < 10)         # True

# 多重赋值
a, b, c = 1, 2, 3
print(a, b, c)            # 1 2 3

# 交换变量
a, b = b, a
print(a, b)               # 2 1

# 解包操作
numbers = [1, 2, 3, 4, 5]
first, *middle, last = numbers
print(first, middle, last)  # 1 [2, 3, 4] 5
```

## 13. 实用示例

```python
# 计算器函数
def calculator(a, b, operation):
    operations = {
        '+': a + b,
        '-': a - b,
        '*': a * b,
        '/': a / b if b != 0 else "错误：除数不能为0",
        '//': a // b if b != 0 else "错误：除数不能为0",
        '%': a % b if b != 0 else "错误：除数不能为0",
        '**': a ** b
    }
    return operations.get(operation, "不支持的运算")

print(calculator(10, 3, '+'))   # 13
print(calculator(10, 3, '//'))  # 3
```
